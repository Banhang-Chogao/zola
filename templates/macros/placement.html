{# ============================================================ Content Placement renderer — render editable content blocks by placement_id. Source of truth: data/content-placements.json (managed by the admin tool at /tools/momo-url/). A "placement" is a stable injection point in a template; a "block" is editable content (title/body/CTA/url) bound to a placement_id. Usage (import once per template, then call at each hook): {% import "macros/placement.html" as placement %} {{ placement::render(id="post_after_intro", page=page | default(value=false)) }} Behaviour (per spec): - No enabled block for the id → renders nothing (build-safe). - Multiple blocks same id → sorted by `priority` ascending. - exclude_pages / pages → prefix-matched against the current path. - HTML is escaped by default (Tera autoescape); only type "html_safe" renders raw markup (admin-only, validated server-side). ============================================================ #} {% macro render(id, page=false, section=false) %} {%- set data = load_data(path="data/content-placements.json", required=false) -%} {%- if data and data.blocks -%} {%- set cur = "" -%} {%- if page and page.path -%}{%- set cur = page.path -%} {%- elif section and section.path -%}{%- set cur = section.path -%}{%- endif -%} {# Scheduling reference point — evaluated once at build time. The site rebuilds on every push plus hourly via Deploy Guard, so start_date/end_date take effect within ~1h of the boundary date, not live per-visitor (static site, no server). #} {%- set now_ts = now(timestamp=true) -%} {%- set_global pl_matched = [] -%} {%- for block in data.blocks -%} {%- if block.placement_id == id and block.enabled -%} {%- set_global pl_ok = true -%} {# Schedule window: start_date/end_date are plain "YYYY-MM-DD" strings. end_date is inclusive through the end of that day. #} {%- if pl_ok and block.start_date -%} {%- set_global cps_start_ts = block.start_date | date(format="%s") | int -%} {%- if now_ts < cps_start_ts -%}{%- set_global pl_ok = false -%}{%- endif -%} {%- endif -%} {%- if pl_ok and block.end_date -%} {%- set_global cps_end_ts = block.end_date | date(format="%s") | int -%} {%- set_global cps_end_ts_excl = cps_end_ts + 86400 -%} {%- if now_ts >= cps_end_ts_excl -%}{%- set_global pl_ok = false -%}{%- endif -%} {%- endif -%} {# exclude_pages: hide when the current path starts with any excluded prefix #} {%- if cur and block.exclude_pages -%} {%- for ex in block.exclude_pages -%} {%- if cur is starting_with(ex) -%}{%- set_global pl_ok = false -%}{%- endif -%} {%- endfor -%} {%- endif -%} {# pages allowlist: "*" = everywhere, else require a prefix match #} {%- if pl_ok and cur and block.pages -%} {%- set_global pl_in = false -%} {%- for pg in block.pages -%} {%- if pg == "*" or cur is starting_with(pg) -%}{%- set_global pl_in = true -%}{%- endif -%} {%- endfor -%} {%- if not pl_in -%}{%- set_global pl_ok = false -%}{%- endif -%} {%- endif -%} {%- if pl_ok -%} {%- set_global pl_matched = pl_matched | concat(with=[block]) -%} {%- endif -%} {%- endif -%} {%- endfor -%} {%- if pl_matched | length > 0 -%} {%- set sorted = pl_matched | sort(attribute="priority") -%}
{%- for block in sorted -%}{{ self::block(block=block) }}{%- endfor -%}
{%- endif -%} {%- endif -%} {% endmacro render %} {# Render a single content block. Autoescape keeps title/body/url safe; only type="html_safe" opts into raw markup (admin-only). #} {% macro block(block) %} {%- set type = block.type | default(value="notice") -%} {%- set style = block.style | default(value="default") -%}
{%- if block.title %}

{{ block.title }}

{% endif -%} {%- if block.body -%} {%- if type == "html_safe" -%}
{{ block.body | safe }}
{%- else -%}

{{ block.body }}

{%- endif -%} {%- endif -%} {%- if block.url and block.button_text -%} {{ block.button_text }} {%- endif -%}
{% endmacro block %}
